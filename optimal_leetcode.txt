int firstMissingPositive(vector<int>& nums) {
        int n = nums.size();
        for(int i = 0; i < n; i++) {
            int& num = nums[i];
            if(num > 0 && num <= n) {
                if(nums[num - 1] != num) {
                    swap(nums[num - 1], num);
                    i--;
                }
            }
        }
        for(int i = 0; i < n; i++) {
            if(nums[i] != i + 1) return i + 1;
        }
        return n + 1;
    }
